Detection queries
Copy directly into BigQuery Console. Replace [PROJECT] with your GCP project ID.
Query 1 — Current period anomaly detection (run nightly)
Schedule as a daily BigQuery job at 06:00. Compares current period DISE totals against rolling baseline and flags 2+ standard deviation anomalies.
anomaly_detection.sql
WITH current_period AS ( SELECT bkpf.MONAT AS fiscal_period, bkpf.GJAHR AS fiscal_year, m.expense_caption, m.dise_category, ROUND(SUM(bseg.DMBTR),0) AS period_total, COUNT(DISTINCT bseg.HKONT) AS gl_account_count, COUNT(*) AS posting_count, COUNT(DISTINCT bkpf.BELNR) AS document_count FROM `[PROJECT].CORTEX_SAP_CDC.bkpf` bkpf JOIN `[PROJECT].CORTEX_SAP_CDC.bseg` bseg ON bkpf.MANDT=bseg.MANDT AND bkpf.BUKRS=bseg.BUKRS AND bkpf.BELNR=bseg.BELNR AND bkpf.GJAHR=bseg.GJAHR JOIN `[PROJECT].dise_reporting.gl_dise_mapping` m ON bseg.HKONT=m.gl_account WHERE bkpf.GJAHR = FORMAT_DATE('%Y',CURRENT_DATE()) AND bkpf.MONAT = FORMAT_DATE('%m',CURRENT_DATE()) AND m.status = 'mapped' AND bkpf.BLART NOT IN ('AA','AF','AB') GROUP BY 1,2,3,4 ), historical_baseline AS ( SELECT expense_caption, dise_category, ROUND(AVG(pa),0) AS avg_amount, ROUND(STDDEV(pa),0) AS stddev_amount, ROUND(MAX(pa),0) AS max_historical, ROUND(MIN(pa),0) AS min_historical, COUNT(DISTINCT period_key) AS periods_in_baseline FROM ( SELECT bkpf2.GJAHR, bkpf2.MONAT, CONCAT(bkpf2.GJAHR,bkpf2.MONAT) AS period_key, m2.expense_caption, m2.dise_category, SUM(bseg2.DMBTR) AS pa FROM `[PROJECT].CORTEX_SAP_CDC.bkpf` bkpf2 JOIN `[PROJECT].CORTEX_SAP_CDC.bseg` bseg2 ON bkpf2.MANDT=bseg2.MANDT AND bkpf2.BUKRS=bseg2.BUKRS AND bkpf2.BELNR=bseg2.BELNR AND bkpf2.GJAHR=bseg2.GJAHR JOIN `[PROJECT].dise_reporting.gl_dise_mapping` m2 ON bseg2.HKONT=m2.gl_account WHERE CONCAT(bkpf2.GJAHR,bkpf2.MONAT) < FORMAT_DATE('%Y%m',CURRENT_DATE()) AND m2.status = 'mapped' AND bkpf2.BLART NOT IN ('AA','AF','AB') GROUP BY 1,2,3,4,5 ) x GROUP BY expense_caption, dise_category ) SELECT c.fiscal_year, c.fiscal_period, c.expense_caption, c.dise_category, c.period_total, h.avg_amount, h.stddev_amount, h.max_historical, h.min_historical, h.periods_in_baseline, c.gl_account_count, c.posting_count, c.document_count, ROUND((c.period_total-h.avg_amount)/NULLIF(h.stddev_amount,0),2) AS std_deviations, ROUND(((c.period_total-h.avg_amount)/NULLIF(h.avg_amount,0))*100,1) AS pct_variance, CASE WHEN ABS((c.period_total-h.avg_amount)/NULLIF(h.stddev_amount,0))>=3 THEN 'CRITICAL' WHEN ABS((c.period_total-h.avg_amount)/NULLIF(h.stddev_amount,0))>=2 THEN 'WARNING' ELSE 'NORMAL' END AS alert_level, CASE WHEN ABS((c.period_total-h.avg_amount)/NULLIF(h.stddev_amount,0))>=3 THEN 2 WHEN ABS((c.period_total-h.avg_amount)/NULLIF(h.stddev_amount,0))>=2 THEN 1 ELSE 0 END AS alert_tier, CURRENT_TIMESTAMP() AS detected_at FROM current_period c JOIN historical_baseline h USING(expense_caption,dise_category) ORDER BY ABS(std_deviations) DESC
Query 2 — Drill-down to source transactions
anomaly_drilldown.sql
SELECT bkpf.BELNR AS document_number, bkpf.BUDAT AS posting_date, bkpf.BLART AS document_type, bkpf.MONAT AS fiscal_period, bseg.HKONT AS gl_account, bseg.KOSTL AS cost_center, bseg.SHKZG AS dr_cr, ROUND(bseg.DMBTR,2) AS amount, m.description AS gl_description, m.dise_category, m.expense_caption FROM `[PROJECT].CORTEX_SAP_CDC.bkpf` bkpf JOIN `[PROJECT].CORTEX_SAP_CDC.bseg` bseg ON bkpf.MANDT=bseg.MANDT AND bkpf.BUKRS=bseg.BUKRS AND bkpf.BELNR=bseg.BELNR AND bkpf.GJAHR=bseg.GJAHR JOIN `[PROJECT].dise_reporting.gl_dise_mapping` m ON bseg.HKONT=m.gl_account WHERE bkpf.GJAHR = FORMAT_DATE('%Y',CURRENT_DATE()) AND bkpf.MONAT = FORMAT_DATE('%m',CURRENT_DATE()) AND m.dise_category = '[CATEGORY]' AND m.expense_caption = '[CAPTION]' AND m.status = 'mapped' AND bkpf.BLART NOT IN ('AA','AF','AB') ORDER BY bkpf.BUDAT DESC, bseg.DMBTR DESC
Query 3 — Acknowledge an alert
acknowledge_alert.sql
UPDATE `[PROJECT].dise_reporting.anomaly_alerts` SET acknowledged = TRUE, acknowledged_by = '[REVIEWER_NAME]', acknowledged_at = CURRENT_TIMESTAMP(), resolution_notes = '[WHAT_CAUSED_THE_ANOMALY]' WHERE fiscal_year = FORMAT_DATE('%Y',CURRENT_DATE()) AND fiscal_period = FORMAT_DATE('%m',CURRENT_DATE()) AND dise_category = '[CATEGORY]' AND expense_caption = '[CAPTION]'
Query 4 — Quarterly digest
quarterly_digest.sql
SELECT fiscal_year, fiscal_period, expense_caption, dise_category, current_amount, historical_avg, ROUND(std_deviations,1) AS std_deviations, ROUND(pct_variance,1) AS pct_variance, alert_level, alert_tier, detected_at, CASE WHEN acknowledged THEN 'Resolved' ELSE 'Outstanding' END AS status, acknowledged_by, acknowledged_at, resolution_notes FROM `[PROJECT].dise_reporting.anomaly_alerts` WHERE fiscal_year = FORMAT_DATE('%Y',CURRENT_DATE()) ORDER BY CASE alert_level WHEN 'CRITICAL' THEN 1 WHEN 'WARNING' THEN 2 ELSE 3 END, ABS(std_deviations) DESC
Production setup
Deploy the alert system to run automatically every day
Step 1 — Create alerts table
create_alerts_table.sql
CREATE TABLE IF NOT EXISTS `[PROJECT].dise_reporting.anomaly_alerts` ( fiscal_year STRING, fiscal_period STRING, expense_caption STRING, dise_category STRING, current_amount FLOAT64, historical_avg FLOAT64, historical_stddev FLOAT64, max_historical FLOAT64, min_historical FLOAT64, periods_in_baseline INT64, gl_account_count INT64, posting_count INT64, document_count INT64, std_deviations FLOAT64, pct_variance FLOAT64, alert_level STRING, alert_tier INT64, detected_at TIMESTAMP, acknowledged BOOL, acknowledged_by STRING, acknowledged_at TIMESTAMP, resolution_notes STRING )
Step 2 — Store Anthropic API key in Secret Manager
Never hardcode API keys. Store in Secret Manager and retrieve at runtime.
secret_manager.sh
echo -n "sk-ant-YOUR_KEY_HERE" | \ gcloud secrets create anthropic-api-key \ --data-file=- --project=[PROJECT] gcloud secrets add-iam-policy-binding anthropic-api-key \ --member="serviceAccount:cortex-sa@[PROJECT].iam.gserviceaccount.com" \ --role="roles/secretmanager.secretAccessor" \ --project=[PROJECT]
